async function submitForm() { // 請記得把這行換成您新部署的 /exec 網址 const scriptURL = 'https://script.google.com/macros/s/AKfycbwVG5nTxjkxxMNssbg4FEdY1Fl4I6e_Z_pk7fFdKNrcp9hP5vhQhs8UG_JaCC2fdhR3/exec'; const data = { warehouse: document.getElementById('warehouse').value, zone: document.getElementById('zone').value, level: document.getElementById('level').value, category: document.getElementById('category').value, description: document.getElementById('description').value }; if (!data.warehouse || !data.zone || !data.description) { alert('請填寫完整資訊！'); return; } const btn = document.querySelector('button'); btn.innerText = '正在傳送至全倉週報庫...'; btn.disabled = true; try { await fetch(scriptURL, { method: 'POST', mode: 'no-cors', body: JSON.stringify(data) }); alert('✅ 提交成功！\n資料已寫入「職安巡檢資料庫」。'); // 清空輸入欄位 document.getElementById('zone').value = ''; document.getElementById('description').value = ''; btn.innerText = '確認提交紀錄'; btn.disabled = false; } catch (error) { alert('傳送失敗，請檢查網路。'); btn.disabled = false; } }
